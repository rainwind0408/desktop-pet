## 二、VRM 骨骼动画修复

### 2.1 现状分析

当前 `index.html` 中 VRM 的 `playMotion` 实现：

```javascript
// 当前代码（仅做表情，无骨骼动画）
function playMotion(name, group) {
    _currentMotion = name;
    fallbackReaction(name);

    if (_threeModel && _threeModel.expressionManager) {
        var expressionMap = { 'happy': 'happy', 'joy': 'happy', 'surprised': 'surprised', 'sad': 'sad', 'upset': 'angry' };
        var expression = expressionMap[name];
        if (expression) {
            _threeModel.expressionManager.setValue(expression, 1.0);
            setTimeout(function() { _threeModel.expressionManager.setValue(expression, 0); }, 1500);
        }
    }
    // ... 无任何骨骼操作
}
```

**问题**：
- 没有使用 `THREE.AnimationMixer`、`THREE.AnimationClip`、`THREE.KeyframeTrack`
- `expressionMap` 只覆盖 5 个名字，InteractionDecider 输出的 `wave`、`nod_fast`、`sleepy`、`scared`、`sit_quietly`、`think` 全部被忽略
- 即使表情名匹配，也只是瞬间切换后 1.5 秒归零，没有过渡动画

### 2.2 修复方案：程序化骨骼动画

由于 VRM 模型不一定自带动画数据，采用 **运行时生成关键帧** 的方式驱动骨骼。

#### 2.2.1 核心函数 `playSkeletalAnimation`

```javascript
/**
 * 程序化骨骼动画播放器
 * @param {Object} vrm - THREE_VRM 实例
 * @param {Array} bones - 骨骼配置数组 [{ name, property, keys }]
 * @param {number} duration - 动画时长（秒）
 */
function playSkeletalAnimation(vrm, bones, duration) {
    if (!_threeMixer) {
        _threeMixer = new THREE.AnimationMixer(vrm.scene);
    }

    var tracks = [];
    bones.forEach(function(boneConfig) {
        var boneNode = vrm.humanoid.getNormalizedBoneNode(boneConfig.name);
        if (!boneNode) return;

        var times = [];
        var values = [];
        boneConfig.keys.forEach(function(key) {
            times.push(key.time);
            values.push(key.x || 0, key.y || 0, key.z || 0, key.w || 1);
        });

        var track = new THREE.QuaternionKeyframeTrack(
            boneNode.name + '.quaternion',
            times,
            values
        );
        tracks.push(track);
    });

    if (tracks.length === 0) return;

    var clip = new THREE.AnimationClip('procedural_' + Date.now(), duration, tracks);
    var action = _threeMixer.clipAction(clip);
    action.setLoop(THREE.LoopOnce);
    action.clampWhenFinished = true;
    action.play();

    // 动画结束后恢复默认姿态
    setTimeout(function() {
        action.stop();
        resetPose(vrm);
    }, duration * 1000);
}
```

#### 2.2.4 改造后的 `playMotion`

```javascript
function playMotion(name, group) {
    _currentMotion = name;
    console.log('Motion:', name, group || '');
    fallbackReaction(name);

    // VRM 骨骼动画
    if (_threeModel && _threeModel.humanoid) {
        var mapping = MOTION_MAP[name];
        if (!mapping) mapping = MOTION_MAP['happy']; // 兜底

        if (mapping.type === 'skeletal' && SKELETAL_ANIMATIONS[mapping.anim]) {
            var anim = SKELETAL_ANIMATIONS[mapping.anim];
            playSkeletalAnimation(_threeModel, anim.bones, anim.duration);
        }
        if (mapping.type === 'expression' && _threeModel.expressionManager) {
            _threeModel.expressionManager.setValue(mapping.name, 1.0);
            setTimeout(function() {
                _threeModel.expressionManager.setValue(mapping.name, 0);
            }, 2000);
        }
    }

    // Live2D（见下一节修复）
    if (_live2dModel) {
        playLive2DMotion(name, group);
    }

    // Bridge 回调
    if (_bridge) {
        _bridge.on_motion_started(group || 'default', name);
        setTimeout(function() { _bridge.on_motion_finished(group || 'default', name); }, 2000);
    }
}
```

#### 3.2.2 改造后的 Live2D 播放函数

```javascript
function playLive2DMotion(name, group) {
    if (!_live2dModel) return;

    var mapping = LIVE2D_MOTION_MAP[name];
    if (!mapping) mapping = LIVE2D_MOTION_MAP['happy'];

    try {
        // pixi-live2d-display 的 motion(group, index) 接口
        _live2dModel.motion(mapping.group, mapping.index);

        console.log('Live2D motion error:', e.message);

        trySetLive2DExpression(name);
    }
}
```

#### 3.2.3 Live2D 表情降级

当 motion 播放失败时（如胡桃无 motion 文件），尝试通过 expression 驱动：

```javascript
function trySetLive2DExpression(name) {
    if (!_live2dModel) return;

    var expressionMap = {
        'happy': 'happy', 'joy': 'happy',
        'sad': 'sad', 'upset': 'sad',
        'surprised': 'surprised', 'scared': 'surprised',
        'angry': 'angry',
        'sleepy': 'relaxed', 'sit_quietly': 'relaxed',
        'think': 'neutral'
    };

    var expr = expressionMap[name];
    if (!expr) return;

    try {
        _live2dModel.expression(expr);
        setTimeout(function() {
            _live2dModel.expression(''); // 重置
        }, 2000);
    } catch(e) {
        console.log('Live2D expression fallback error:', e.message);
    }
}
```

## 四、纯表情模型支持（胡桃专项）

### 4.1 问题

胡桃的 `Hu Tao.model3.json` 中：
- 无 `Motions` 字段
- 仅有 `Expressions`（8 个 `.exp3.json` 文件）
- 上述 Live2D motion 调用必然失败

### 4.2 修复方案：EXPRESSION_ONLY_MAP

对于无 motion 文件的 Live2D 模型，所有动画名直接映射到表情：

```javascript
var EXPRESSION_ONLY_MAP = {
    'wave':        'happy',
    'nod_fast':    'happy',
    'shake':       'surprised',
    'bow':         'happy',
    'wave_arm':    'happy',
    'greet':       'happy',
    'happy':       'happy',
    'sad':         'sad',
    'angry':       'angry',
    'surprised':   'surprised',
    'sleepy':      'relaxed',
    'scared':      'surprised',
    'sit_quietly': 'relaxed',
    'think':       'neutral'
};
```

在 `playLive2DMotion` 中增加检测逻辑：

```javascript
function playLive2DMotion(name, group) {
    if (!_live2dModel) return;

    // 检测是否为纯表情模型（无 motion 定义）
    var isExpressionOnly = !_live2dModel.internalModel.motionManager
        || !_live2dModel.internalModel.motionManager.definitions
        || Object.keys(_live2dModel.internalModel.motionManager.definitions).length === 0;

    if (isExpressionOnly) {
        var expr = EXPRESSION_ONLY_MAP[name] || 'happy';
        try {
            _live2dModel.expression(expr);
            setTimeout(function() { _live2dModel.expression(''); }, 2500);
        } catch(e) {
            console.log('Expression-only fallback error:', e.message);
        }
        return;
    }

    // 正常 motion 播放
    var mapping = LIVE2D_MOTION_MAP[name];
    if (!mapping) mapping = LIVE2D_MOTION_MAP['happy'];
    try {
        _live2dModel.motion(mapping.group, mapping.index);
    } catch(e) {
        trySetLive2DExpression(name);
    }
}
```

## 八、风险与注意事项

1. **骨骼名称兼容性**：不同 VRM 模型的骨骼节点名可能不同（如 `leftUpperArm` vs `LeftUpperArm`），`playSkeletalAnimation` 中已使用 `getNormalizedBoneNode` 做归一化，但仍需验证。

2. **四元数旋转**：骨骼动画使用四元数（Quaternion），不能直接用欧拉角。预定义动画中的 `x/y/z/w` 值需要正确转换为四元数分量。上述代码简化处理，实际实现时可能需要 `THREE.Quaternion().setFromEuler(new THREE.Euler(x, y, z))`。

3. **AnimationMixer 冲突**：如果多个骨骼动画同时触发（快速点击），需要处理 action 混合或强制停止上一个动画。

4. **Live2D motion index**：不同 Live2D 模型的 motion group 和 index 布局不同，映射表可能需要按模型调整。建议在模型配置文件中增加 `motionMap` 字段允许覆盖默认映射。

5. **性能**：程序化骨骼动画每帧更新，CPU 开销很低。但需要确保 `animate()` 中 mixer.update 的调用在 renderer.render 之前。

---

## 九、后续优化方向

1. **动画配置外部化**：将 `SKELETAL_ANIMATIONS`、`MOTION_MAP` 等提取到角色的 `profile.json` 或独立配置文件中，允许每个角色自定义动画映射。

2. **动画队列**：支持动画排队播放，避免快速点击时动画互相覆盖。

3. **过渡动画**：在表情/动作切换时增加淡入淡出过渡，避免瞬间跳变。

4. **Live2D motion 文件补充**：为胡桃等纯表情角色制作 `.motion3.json` 文件，丰富动画表现。

5. **Sprite 动画扩展**：为 sprite 角色增加更多动画行（如 think、sleepy 独立行），减少降级复用。
